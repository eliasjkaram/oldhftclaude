```python
import requests

def call_openrouter_llm(prompt, model="google/gemini-2.5-pro"):
    response = requests.post(
        "https://openrouter.ai/api/v1/chat/completions",
        headers={
            "Authorization": f"Bearer {OPENROUTER_API_KEY}",
            "HTTP-Referer": "https://github.com/alpaca-mcp",
            "X-Title": "Alpaca Trading System"
        },
        json={
            "model": model,
            "messages": [{"role": "user", "content": prompt}]
        }
    )
    return response.json()
```
